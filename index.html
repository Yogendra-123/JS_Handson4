                {
                  const B = [];
                  let even= 0;
                  let odd = 0;
                  for (i=0; i< Arr.length; i++){
                    if (Arr[i]%2===0){
                      even += Arr[i];
                      B[0] = even;
                    }else{
                      odd += Arr[i];
                      B[1] = odd;
                    }
                  }
                  return B;
                };
